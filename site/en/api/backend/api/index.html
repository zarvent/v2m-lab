    async def broadcast_event(self, event_type: str, data: dict[str, Any]) -> None:
    """Env√≠a evento a todos los clientes WebSocket conectados.

    Args:
        event_type: Tipo de evento (ej. 'transcription_update', 'heartbeat').
        data: Payload del evento como diccionario JSON-serializable.

    Note:
        Las conexiones muertas se limpian autom√°ticamente.
    """
    if not self._websocket_clients:
        return

    message = {"event": event_type, "data": data}
    disconnected: list[WebSocket] = []

    for ws in self._websocket_clients:
        try:
            await ws.send_json(message)
        except Exception:
            disconnected.append(ws)

    # Cleanup de conexiones muertas
    for ws in disconnected:
        self._websocket_clients.discard(ws)
